var mongoose = require('mongoose');
var City = mongoose.model('City');
 
// Add city to request
exports.city = function(req, res, next, id) {
  City.load(id, function(err, city) {
    if (err){
      return next(err);
    }
 
    if (!city){
      return next(new Error('Failed to load city ' + id));
    }
    req.city = city;
    next();
  });
};
